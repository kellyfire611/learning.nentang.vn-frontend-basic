5.1 Business test là gì?

Business test (hay domain test) kiểm tra các quy tắc nghiệp vụ: giảm giá theo cấp độ, tính phí vận chuyển, áp dụng voucher, kiểm tra điều kiện mua hàng…

Khác với unit test kiểm tra hàm thuần túy, business test kiểm tra logic thể hiện yêu cầu từ phía business — thay đổi yêu cầu thì test phải cập nhật theo.

5.2 Ví dụ: Hàm tính giảm giá theo bậc

// src/js/pricing.js
export function calculateDiscount(subtotal) {
  if (subtotal >= 1000000) return subtotal * 0.15;
  if (subtotal >= 500000)  return subtotal * 0.10;
  if (subtotal >= 200000)  return subtotal * 0.05;
  return 0;
}

export function getShippingFee(subtotal) {
  return subtotal >= 500000 ? 0 : 30000;
}

5.3 Viết business test

// tests/business/pricing.test.js
import { describe, it, expect } from 'vitest';
import { calculateDiscount, getShippingFee } from '../../src/js/pricing.js';

describe('calculateDiscount — quy tắc giảm giá', () => {
  it('giảm 15% khi tổng >= 1,000,000', () => {
    expect(calculateDiscount(1000000)).toBe(150000);
  });

  it('giảm 10% khi tổng >= 500,000', () => {
    expect(calculateDiscount(500000)).toBe(50000);
  });

  it('giảm 5% khi tổng >= 200,000', () => {
    expect(calculateDiscount(200000)).toBe(10000);
  });

  it('không giảm khi tổng < 200,000', () => {
    expect(calculateDiscount(199999)).toBe(0);
  });

  // Boundary cases
  it('đúng ngưỡng 500,000: áp dụng mức 10% chứ không phải 15%', () => {
    expect(calculateDiscount(500000)).toBe(50000);
    expect(calculateDiscount(999999)).toBe(99999.9);
  });
});

describe('getShippingFee — phí vận chuyển', () => {
  it('miễn phí khi tổng >= 500,000', () => {
    expect(getShippingFee(500000)).toBe(0);
    expect(getShippingFee(600000)).toBe(0);
  });

  it('tính phí 30,000 khi tổng < 500,000', () => {
    expect(getShippingFee(499999)).toBe(30000);
    expect(getShippingFee(0)).toBe(30000);
  });
});
Best practice: Đặt tên test bằng tiếng Việt rõ ràng theo quy tắc nghiệp vụ. Khi test fail, chỉ cần đọc tên là biết ngay quy tắc nào bị vi phạm.
