4.1 Tách hàm để dễ test

Trước khi viết test, bạn cần đảm bảo logic có thể import được. Tách logic ra khỏi HTML và export: 

// src/js/cart.js
export function addToCart(cart, item) {
  if (!item || item.qty <= 0) throw new Error('Số lượng không hợp lệ');
  if (item.price < 0) throw new Error('Giá không hợp lệ');
  const existing = cart.find(i => i.id === item.id);
  if (existing) {
    existing.qty += item.qty;
  } else {
    cart.push({ ...item });
  }
  return cart;
}

export function getTotal(cart) {
  return cart.reduce((sum, item) => sum + item.price * item.qty, 0);
}

4.2 Viết unit test với Vitest

// tests/unit/cart.test.js
import { describe, it, expect } from 'vitest';
import { addToCart, getTotal } from '../../src/js/cart.js';

describe('addToCart', () => {
  it('thêm sản phẩm mới vào giỏ rỗng', () => {
    const cart = [];
    const item = { id: 1, name: 'Áo', price: 100000, qty: 1 };
    addToCart(cart, item);
    expect(cart).toHaveLength(1);
    expect(cart[0].name).toBe('Áo');
  });

  it('cộng dồn số lượng nếu sản phẩm đã có trong giỏ', () => {
    const cart = [{ id: 1, name: 'Áo', price: 100000, qty: 1 }];
    addToCart(cart, { id: 1, name: 'Áo', price: 100000, qty: 2 });
    expect(cart[0].qty).toBe(3);
  });

  it('ném lỗi khi qty <= 0', () => {
    expect(() => addToCart([], { id: 1, price: 100000, qty: 0 }))
      .toThrow('Số lượng không hợp lệ');
  });
});

describe('getTotal', () => {
  it('tính tổng đúng với nhiều sản phẩm', () => {
    const cart = [
      { id: 1, price: 100000, qty: 2 },
      { id: 2, price: 50000, qty: 1 },
    ];
    expect(getTotal(cart)).toBe(250000);
  });

  it('trả về 0 với giỏ rỗng', () => {
    expect(getTotal([])).toBe(0);
  });
});

4.3 Chạy test và xem coverage

# chạy một lần
npm test

# chạy watch mode (tự reload khi lưu file)
npm run test:watch

# xem coverage report
npx vitest run --coverage
Lỗi thường gặp: Cannot use import statement outside a module — thêm "type": "module" vào package.json hoặc cấu hình Vitest dùng transform. Vitest thường tự xử lý nếu dùng đúng config.
