9.2.3 Ham fetchCustomers

Ham nay nhan fetchFn = fetch de de mock trong unit test.

export async function fetchCustomers(fetchFn = fetch) {
  const response = await fetchFn("https://jsonplaceholder.typicode.com/users");

  if (!response.ok) {
    throw new Error("Khong the tai danh sach khach hang!");
  }

  return response.json();
}
