10.8.1 Business test khac unit test o dau?

Unit test hoi: ham co tra dung gia tri khong. Business test hoi: tinh huong nguoi dung can co duoc dap ung khong. Cach viet nen doc theo mau Cho truoc - Khi - Thi.

10.8.2 Code day du vi du business test

import { describe, expect, it, vi } from "vitest";
import { createTodo } from "../../src/todoApiStore.js";
import {
  createTodoPayload, filterTodos, removeTodoFromList,
  updateTodoInList, validateTodoTitle,
} from "../../src/todoBusiness.js";

describe("Business test Todo theo Cho truoc - Khi - Thi", () => {
  it("tao payload tu ten cong viec hop le", () => {
    // Cho truoc
    const title = "Hoc JSON Server";
    // Khi
    const payload = createTodoPayload(title);
    // Thi
    expect(payload).toEqual({ title: "Hoc JSON Server", completed: false });
  });

  it("khong cho them khi ten cong viec rong", () => {
    const errorMessage = validateTodoTitle("   ");
    expect(errorMessage).toBe("Vui long nhap ten cong viec!");
  });

  it("loc cong viec chua hoan thanh", () => {
    const todos = [
      { id: "1", title: "Da xong", completed: true },
      { id: "2", title: "Chua xong", completed: false },
    ];
    expect(filterTodos(todos, "active")).toEqual([
      { id: "2", title: "Chua xong", completed: false },
    ]);
  });

  it("API tra ve thanh cong khi them Todo hop le", async () => {
    const payload = createTodoPayload("Hoc Vitest");
    const fetchFn = vi.fn().mockResolvedValue({
      ok: true,
      json: async () => ({ id: "10", ...payload }),
    });
    const newTodo = await createTodo(payload, fetchFn);
    expect(newTodo.id).toBe("10");
  });
});
