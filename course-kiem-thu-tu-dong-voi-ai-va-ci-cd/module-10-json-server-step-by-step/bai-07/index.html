10.7.1 Test logic thuan

import { describe, expect, it } from "vitest";
import {
  createTodoPayload, filterTodos, removeTodoFromList,
  updateTodoInList, validateTodoTitle,
} from "../../src/todoBusiness.js";

describe("Unit test nghiep vu Todo", () => {
  it("bao loi khi title rong", () => {
    expect(validateTodoTitle("   ")).toBe("Vui long nhap ten cong viec!");
  });

  it("tao payload da trim va chua hoan thanh", () => {
    expect(createTodoPayload("  Hoc API  ")).toEqual({
      title: "Hoc API", completed: false,
    });
  });

  it("loc Todo active", () => {
    const todos = [
      { id: "1", completed: true },
      { id: "2", completed: false },
    ];
    expect(filterTodos(todos, "active")).toEqual([{ id: "2", completed: false }]);
  });
});

10.7.2 Mock fetch de test API

import { describe, expect, it, vi } from "vitest";
import { TODO_API_URL, createTodo, deleteTodo, fetchTodos } from "../../src/todoApiStore.js";

function successFetch(data) {
  return vi.fn().mockResolvedValue({ ok: true, json: async () => data });
}

describe("Unit test API Todo", () => {
  it("GET dung endpoint", async () => {
    const fetchFn = successFetch([]);
    await fetchTodos(fetchFn);
    expect(fetchFn).toHaveBeenCalledWith(TODO_API_URL);
  });

  it("POST dung method va body", async () => {
    const todo = { title: "Hoc API", completed: false };
    const fetchFn = successFetch({ id: "3", ...todo });
    await createTodo(todo, fetchFn);
    expect(fetchFn.mock.calls[0][1].method).toBe("POST");
    expect(fetchFn.mock.calls[0][1].body).toBe(JSON.stringify(todo));
  });

  it("DELETE dung URL co id", async () => {
    const fetchFn = successFetch({});
    await deleteTodo("99", fetchFn);
    expect(fetchFn.mock.calls[0][0]).toBe(`${TODO_API_URL}/99`);
  });
});
