10.10.2 Code E2E mau

import { expect, test } from "@playwright/test";

const API_URL = "http://localhost:3001/todos";
const TEST_PREFIX = "E2E Todo";

async function createTodoByApi(request, title, completed = false) {
  const response = await request.post(API_URL, { data: { title, completed } });
  return response.json();
}

async function cleanupE2eTodos(request) {
  const response = await request.get(API_URL);
  const todos = await response.json();
  for (const todo of todos.filter((todo) => todo.title.startsWith(TEST_PREFIX))) {
    await request.delete(`${API_URL}/${todo.id}`);
  }
}

test.afterEach(async ({ request }) => cleanupE2eTodos(request));

test("mo trang va tai danh sach Todo", async ({ page }) => {
  await page.goto("/");
  await expect(page.locator("h1")).toContainText("Quan ly cong viec Todo");
  await expect(page.locator('[data-testid="todo-item"]').first()).toBeVisible();
});

test("them Todo moi thanh cong", async ({ page }) => {
  await page.goto("/");
  await page.locator("#todo-title-input").fill(`${TEST_PREFIX} them moi`);
  await page.locator("#add-todo-button").click();
  await expect(page.locator("#todo-list")).toContainText(`${TEST_PREFIX} them moi`);
});

test("sua ten Todo", async ({ page, request }) => {
  const todo = await createTodoByApi(request, `${TEST_PREFIX} can sua`, false);
  await page.goto("/");
  page.on("dialog", async (dialog) => dialog.accept(`${TEST_PREFIX} da sua`));
  await page.locator(`[data-id="${todo.id}"]`).locator('[data-testid="edit-todo-button"]').click();
  await expect(page.locator("#todo-list")).toContainText(`${TEST_PREFIX} da sua`);
});
