📚 Đồ Án 7: Website Truyện Tranh Online

Module 4 – Đồ Án Tổng Hợp PHP

📋 Mô Tả Dự Án

Xây dựng website đọc truyện tranh online hoàn chỉnh với hệ thống chapter ảnh, lượt xem, bình luận, yêu thích cho người dùng và trang Admin quản lý toàn bộ nội dung. Database thiết kế đơn giản gồm 3 bảng lõi: truyen, chuong, chuong_hinhanh.

🗄️ Database Schema (3 bảng lõi + mở rộng)

  • truyen – thông tin truyện: tên, slug, ảnh bìa, tác giả, thể loại, mô tả, trạng thái, lượt xem
  • chuong – mỗi chapter: truyen_id, số thứ tự, tiêu đề, ngày đăng
  • chuong_hinhanh – mỗi ảnh trong chapter: chuong_id, đường dẫn file, thứ tự trang
  • the_loai & truyen_the_loai – many-to-many thể loại (Action, Romance, Fantasy…)
  • users – tài khoản người đọc (đăng nhập để yêu thích, bình luận)
  • yeu_thich – user_id + truyen_id (unique)
  • binh_luan – bình luận theo chapter, hỗ trợ reply lồng nhau
  • lich_su_doc – lưu chapter cuối user đã đọc mỗi bộ truyện

✅ Tính năng phía Người Dùng (Frontend)

  • Trang chủ: truyện mới cập nhật, truyện hot (lượt xem cao), truyện theo thể loại
  • Tìm kiếm: tìm theo tên, lọc theo thể loại, trạng thái (đang ra / hoàn thành), sắp xếp theo lượt xem / ngày cập nhật
  • Trang chi tiết truyện: ảnh bìa, mô tả, danh sách chapter với ngày đăng, nút Đọc Mới Nhất / Đọc Từ Đầu
  • Trang đọc chapter: hiển thị ảnh full-width tuần tự, điều hướng chapter trước/sau, lazy-load ảnh
  • Lịch sử đọc: tự động lưu chapter cuối, tiếp tục đọc từ chỗ dừng
  • Yêu thích: thêm/bỏ yêu thích (cần đăng nhập), trang quản lý danh sách yêu thích
  • Bình luận: bình luận theo chapter, reply, phân trang bình luận
  • Thể loại: trang liệt kê truyện theo từng tag thể loại

🛠️ Tính năng Admin Panel

  • Dashboard: tổng số truyện, chapter, lượt xem, người dùng, biểu đồ top 10 truyện hot
  • Quản lý Truyện: thêm/sửa/xóa truyện (tên, mô tả, ảnh bìa, thể loại, trạng thái)
  • Quản lý Chapter: thêm chapter mới, sửa tiêu đề, xóa chapter và toàn bộ ảnh
  • Upload Ảnh: upload nhiều ảnh cùng lúc, sắp xếp thứ tự trang bằng drag-and-drop, xóa từng trang
  • Quản lý Thể Loại: CRUD thể loại, gắn thể loại cho truyện
  • Quản lý Bình Luận: duyệt / xóa bình luận vi phạm
  • Quản lý Users: xem danh sách, khóa/mở tài khoản

🧠 Kiến thức áp dụng

  • Slug URL thân thiện: /truyen/one-piece/chuong-1100
  • Lazy-load ảnh chapter: dùng loading="lazy" hoặc IntersectionObserver JS
  • Upload nhiều file: <input type="file" multiple> + vòng lặp $_FILES
  • Tăng lượt xem không bị F5 spam: dùng Session hoặc cookie đánh dấu đã xem
  • Truy vấn tối ưu: dùng JOIN lấy chapter mới nhất cùng lúc với query danh sách truyện
  • Many-to-many thể loại: bảng pivot + GROUP_CONCAT trong SQL
  • Phân quyền: middleware kiểm tra $_SESSION['role'] === 'admin'
  • Xóa cascade: khi xóa truyện → xóa chapter → xóa ảnh (file + DB)

// ══ TRUYEN REPOSITORY ════════════════
class TruyenRepository {
    public function __construct(private PDO $pdo) {}

    // ─── Trang chủ: truyện mới cập nhật ─
    public function getMoiCapNhat(int $limit = 20): array {
        return $this->pdo->query("
            SELECT t.id, t.ten, t.slug, t.anh_bia, t.trang_thai,
                   MAX(c.so_thu_tu) AS chuong_moi,
                   MAX(c.created_at) AS cap_nhat_luc
            FROM truyen t
            LEFT JOIN chuong c ON c.truyen_id = t.id
            GROUP BY t.id
            ORDER BY cap_nhat_luc DESC
            LIMIT $limit
        ")->fetchAll();
    }

    // ─── Top truyện hot (lượt xem cao) ──
    public function getTopHot(int $limit = 10, string $khoang = 'week'): array {
        $dk_ngay = match($khoang) {
            'day'   => "AND c.created_at >= DATE_SUB(NOW(), INTERVAL 1 DAY)",
            'week'  => "AND c.created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)",
            'month' => "AND c.created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)",
            default => ""
        };
        return $this->pdo->query("
            SELECT t.id, t.ten, t.slug, t.anh_bia, SUM(c.luot_xem) AS tong_xem
            FROM truyen t
            JOIN chuong c ON c.truyen_id = t.id
            WHERE 1=1 $dk_ngay
            GROUP BY t.id
            ORDER BY tong_xem DESC
            LIMIT $limit
        ")->fetchAll();
    }

    // ─── Chi tiết truyện kèm thể loại ───
    public function getBySlug(string $slug): ?array {
        $stmt = $this->pdo->prepare("
            SELECT t.*,
                   GROUP_CONCAT(tl.ten ORDER BY tl.ten SEPARATOR ', ') AS the_loai,
                   GROUP_CONCAT(tl.slug ORDER BY tl.ten SEPARATOR ',') AS the_loai_slugs,
                   COUNT(DISTINCT c.id) AS so_chuong
            FROM truyen t
            LEFT JOIN truyen_the_loai ttl ON ttl.truyen_id = t.id
            LEFT JOIN the_loai tl ON tl.id = ttl.the_loai_id
            LEFT JOIN chuong c ON c.truyen_id = t.id
            WHERE t.slug = ?
            GROUP BY t.id
        ");
        $stmt->execute([$slug]);
        return $stmt->fetch() ?: null;
    }

    // ─── Danh sách chương của truyện ────
    public function getDanhSachChuong(int $truyen_id): array {
        $stmt = $this->pdo->prepare("
            SELECT id, so_thu_tu, tieu_de, luot_xem, created_at
            FROM chuong WHERE truyen_id = ?
            ORDER BY so_thu_tu DESC
        ");
        $stmt->execute([$truyen_id]);
        return $stmt->fetchAll();
    }

    // ─── Tìm kiếm nâng cao ───────────────
    public function search(array $filter, int $page = 1, int $limit = 24): array {
        $where = ["1=1"];
        $params = [];

        if (!empty($filter['q'])) {
            $where[] = "(t.ten LIKE ? OR t.tac_gia LIKE ?)";
            $params[] = "%{$filter['q']}%";
            $params[] = "%{$filter['q']}%";
        }
        if (!empty($filter['the_loai'])) {
            $where[] = "EXISTS (
                SELECT 1 FROM truyen_the_loai ttl
                JOIN the_loai tl ON tl.id = ttl.the_loai_id
                WHERE ttl.truyen_id = t.id AND tl.slug = ?
            )";
            $params[] = $filter['the_loai'];
        }
        if (!empty($filter['trang_thai'])) {
            $where[] = "t.trang_thai = ?";
            $params[] = $filter['trang_thai'];
        }

        $order = match($filter['sap_xep'] ?? 'moi') {
            'xem'   => "t.luot_xem DESC",
            'ten'   => "t.ten ASC",
            default => "t.updated_at DESC",
        };

        $where_sql = implode(' AND ', $where);
        $offset     = ($page - 1) * $limit;

        $total = $this->pdo->prepare(
            "SELECT COUNT(*) FROM truyen t WHERE $where_sql"
        );
        $total->execute($params);

        $rows = $this->pdo->prepare("
            SELECT t.id, t.ten, t.slug, t.anh_bia, t.trang_thai, t.luot_xem,
                   MAX(c.so_thu_tu) AS chuong_moi
            FROM truyen t
            LEFT JOIN chuong c ON c.truyen_id = t.id
            WHERE $where_sql
            GROUP BY t.id
            ORDER BY $order
            LIMIT $limit OFFSET $offset
        ");
        $rows->execute($params);

        return [
            'items' => $rows->fetchAll(),
            'total' => (int)$total->fetchColumn(),
            'page'  => $page,
            'pages' => (int)ceil($total->fetchColumn() / $limit),
        ];
    }

    // ─── Tăng lượt xem (chống F5 spam) ──
    public function tangLuotXem(int $truyen_id, int $chuong_id): void {
        $key = "viewed_{$chuong_id}";
        if (!isset($_SESSION[$key])) {
            $_SESSION[$key] = true;
            $this->pdo->prepare("UPDATE chuong SET luot_xem = luot_xem + 1 WHERE id=?")
                 ->execute([$chuong_id]);
            $this->pdo->prepare("UPDATE truyen SET luot_xem = luot_xem + 1 WHERE id=?")
                 ->execute([$truyen_id]);
        }
    }
}

// ══ CHUONG REPOSITORY ════════════════
class ChuongRepository {
    public function __construct(private PDO $pdo) {}

    public function getWithImages(int $chuong_id): ?array {
        $stmt = $this->pdo->prepare("SELECT * FROM chuong WHERE id = ?");
        $stmt->execute([$chuong_id]);
        $chuong = $stmt->fetch();
        if (!$chuong) return null;

        $imgs = $this->pdo->prepare(
            "SELECT duong_dan FROM chuong_hinhanh WHERE chuong_id = ? ORDER BY thu_tu"
        );
        $imgs->execute([$chuong_id]);
        $chuong['hinhanh'] = $imgs->fetchAll(PDO::FETCH_COLUMN);
        return $chuong;
    }

    // ─── Chapter trước / sau ─────────────
    public function getNavigation(int $truyen_id, float $so_thu_tu): array {
        $prev = $this->pdo->prepare(
            "SELECT id, so_thu_tu FROM chuong WHERE truyen_id=? AND so_thu_tu < ? ORDER BY so_thu_tu DESC LIMIT 1"
        );
        $prev->execute([$truyen_id, $so_thu_tu]);

        $next = $this->pdo->prepare(
            "SELECT id, so_thu_tu FROM chuong WHERE truyen_id=? AND so_thu_tu > ? ORDER BY so_thu_tu ASC LIMIT 1"
        );
        $next->execute([$truyen_id, $so_thu_tu]);

        return ['prev' => $prev->fetch() ?: null, 'next' => $next->fetch() ?: null];
    }

    // ─── Upload ảnh cho chapter ──────────
    public function uploadHinhanh(int $chuong_id, array $files, string $upload_dir): int {
        $inserted = 0;
        $stmt_max  = $this->pdo->prepare(
            "SELECT COALESCE(MAX(thu_tu),0) FROM chuong_hinhanh WHERE chuong_id=?"
        );
        $stmt_max->execute([$chuong_id]);
        $thu_tu = (int)$stmt_max->fetchColumn();

        $stmt = $this->pdo->prepare(
            "INSERT INTO chuong_hinhanh (chuong_id, duong_dan, thu_tu) VALUES (?,?,?)"
        );

        foreach ($files['tmp_name'] as $i => $tmp) {
            if ($files['error'][$i] !== UPLOAD_ERR_OK) continue;
            $ext  = strtolower(pathinfo($files['name'][$i], PATHINFO_EXTENSION));
            if (!in_array($ext, ['jpg','jpeg','png','webp','gif'])) continue;
            $thu_tu++;
            $filename  = sprintf('%04d.%s', $thu_tu, $ext);
            $dest_dir  = rtrim($upload_dir, '/') . "/$chuong_id/";
            if (!is_dir($dest_dir)) mkdir($dest_dir, 0755, true);
            if (move_uploaded_file($tmp, $dest_dir . $filename)) {
                $stmt->execute([$chuong_id, "uploads/chuong/$chuong_id/$filename", $thu_tu]);
                $inserted++;
            }
        }
        return $inserted;
    }
}

// ══ YEU THICH SERVICE ════════════════
class YeuThichService {
    public function __construct(private PDO $pdo) {}

    public function toggle(int $user_id, int $truyen_id): bool {
        $check = $this->pdo->prepare(
            "SELECT 1 FROM yeu_thich WHERE user_id=? AND truyen_id=?"
        );
        $check->execute([$user_id, $truyen_id]);
        if ($check->fetch()) {
            $this->pdo->prepare("DELETE FROM yeu_thich WHERE user_id=? AND truyen_id=?")
                 ->execute([$user_id, $truyen_id]);
            return false; // đã bỏ yêu thích
        }
        $this->pdo->prepare("INSERT INTO yeu_thich (user_id, truyen_id) VALUES (?,?)")
             ->execute([$user_id, $truyen_id]);
        return true;  // đã thêm yêu thích
    }

    public function getDanhSach(int $user_id): array {
        $stmt = $this->pdo->prepare("
            SELECT t.id, t.ten, t.slug, t.anh_bia, t.trang_thai,
                   MAX(c.so_thu_tu) AS chuong_moi,
                   lsd.chuong_id AS chuong_dang_doc
            FROM yeu_thich yt
            JOIN truyen t ON t.id = yt.truyen_id
            LEFT JOIN chuong c ON c.truyen_id = t.id
            LEFT JOIN lich_su_doc lsd ON lsd.user_id = ? AND lsd.truyen_id = t.id
            WHERE yt.user_id = ?
            GROUP BY t.id
            ORDER BY yt.created_at DESC
        ");
        $stmt->execute([$user_id, $user_id]);
        return $stmt->fetchAll();
    }
}

// ══ ADMIN – XÓA TRUYỆN CASCADE ═══════
function adminXoaTruyen(PDO $pdo, int $truyen_id, string $upload_dir): void {
    // Lấy tất cả ảnh trước khi xóa
    $hinh = $pdo->prepare("
        SELECT ch.duong_dan FROM chuong_hinhanh ch
        JOIN chuong c ON c.id = ch.chuong_id
        WHERE c.truyen_id = ?
    ");
    $hinh->execute([$truyen_id]);
    $files = $hinh->fetchAll(PDO::FETCH_COLUMN);

    // Xóa DB (CASCADE tự xóa chuong + chuong_hinhanh + lich_su_doc, yeu_thich)
    $pdo->prepare("DELETE FROM truyen WHERE id=?")->execute([$truyen_id]);

    // Xóa file vật lý
    foreach ($files as $path) {
        $full = rtrim($upload_dir, '/') . '/' . ltrim($path, '/');
        if (file_exists($full)) unlink($full);
    }
    // Dọn thư mục rỗng
    $dir = $upload_dir . "/chuong/";
    foreach (glob("$dir*", GLOB_ONLYDIR) as $d) {
        if (count(glob("$d/*")) === 0) rmdir($d);
    }
}
