Bai 08: Flexbox Container co ban

Module 3 CSS Layout & Responsive

/* Kich hoat Flexbox */
.container { display: flex; }

/* Huong sap xep (Main Axis) */
.hang   { flex-direction: row; }    /* mac dinh, trai -> phai */
.cot    { flex-direction: column; } /* tren -> duoi */
.hang-ngược { flex-direction: row-reverse; }
.cot-nguoc  { flex-direction: column-reverse; }

/* Xuong dong khi het cho */
.tu-dong { flex-wrap: wrap; }    /* tu dong xuong dong */
.mot-dong { flex-wrap: nowrap; }  /* mac dinh, khong xuong dong */

/* Shorthand */
.flex { flex-flow: row wrap; } /* direction + wrap */

/* Khoang cach giua items */
.gap { gap: 16px; }          /* tat ca huong */
.gap-xy { row-gap: 16px; column-gap: 24px; }
