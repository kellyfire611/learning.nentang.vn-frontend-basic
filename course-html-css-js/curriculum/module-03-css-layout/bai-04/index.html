Bai 04: Position relative va absolute

Module 3 CSS Layout & Responsive

/* relative - di chuyen tuong doi vi tri ban dau */
.hop { position: relative; top: 20px; left: 30px; }
/* Nhung van chiem khong gian cu */

/* absolute - dinh vi theo cha gan nhat co position */
.cha { position: relative; } /* lam moc cho con */
.con { position: absolute; top: 10px; right: 10px; }

/* Ung dung thuc te: Badge tren anh */
.san-pham { position: relative; }
.badge {
  position: absolute;
  top: 8px; right: 8px;
  background: red; color: white;
  padding: 4px 8px; border-radius: 10px;
}

/* Ung dung: Can giua tuyet doi */
.trung-tam {
  position: absolute;
  top: 50%; left: 50%;
  transform: translate(-50%, -50%);
}
