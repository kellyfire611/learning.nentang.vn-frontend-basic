Bai 22: Media Queries co ban

/* Diem ngat pho bien */
/* Mobile:  < 640px  */
/* Tablet:  640-1024px */
/* Desktop: > 1024px */

/* Desktop-first (viet cho desktop truoc, thu nho lai) */
.container { width: 1200px; }
@media (max-width: 1024px) { .container { width: 100%; padding: 0 20px; } }
@media (max-width: 640px)  { .container { padding: 0 16px; } }

/* Ket hop dieu kien */
@media (min-width: 640px) and (max-width: 1024px) {
  /* Chi ap dung cho tablet */
}

/* Theo huong man hinh */
@media (orientation: landscape) { /* man hinh ngang */ }
@media (orientation: portrait)  { /* man hinh doc */ }

/* Chieu cao */
@media (max-height: 500px) { .hero { height: auto; padding: 40px; } }

/* In an */
@media print { .no-print { display: none; } }
