Bai 12: Flexbox order va align-self

/* order: thay doi thu tu hien thi (mac dinh = 0) */
.vat-1 { order: 3; } /* hien thi sau cung */
.vat-2 { order: 1; } /* hien thi dau tien */
.vat-3 { order: 2; } /* hien thi giua */

/* align-self: ghi de align-items cho TUNG item */
.container { display: flex; align-items: center; }
.item-dac-biet { align-self: flex-end; } /* item nay o cuoi */
.item-top      { align-self: flex-start; }
.item-gian     { align-self: stretch; }

/* Ung dung: Pin footer xuong cuoi */
.page { display: flex; flex-direction: column; min-height: 100vh; }
.main-content { flex: 1; } /* chiem phan con lai */
.footer { /* tu dong xuong cuoi */ }
