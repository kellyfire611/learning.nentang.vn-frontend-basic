Bai 20: CSS Grid auto-fill va auto-fit

/* auto-fill: tao cot rong ngay ca khi khong co item */
.auto-fill {
  display: grid;
  grid-template-columns: repeat(auto-fill, minmax(200px, 1fr));
}

/* auto-fit: keo gian cot con lai khi it item */
.auto-fit {
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(200px, 1fr));
}

/* Khac biet: auto-fill giu cot trong, auto-fit bo cot trong */

/* Pattern hay dung nhat cho responsive card */
.card-grid {
  display: grid;
  grid-template-columns: repeat(auto-fill, minmax(280px, 1fr));
  gap: 24px;
}
/* Hoan toan responsive KHONG can media queries! */
