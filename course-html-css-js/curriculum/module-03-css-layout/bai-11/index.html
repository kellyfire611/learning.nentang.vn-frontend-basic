Trang Chu Module 3 Bai 11

Bai 11: Flexbox flex-grow shrink basis

.container { display: flex; }

/* flex-grow: ty le phong to khi co khong gian thua */
.col-1 { flex-grow: 1; } /* phan thua chia deu */
.col-2 { flex-grow: 2; } /* nhan gap doi phan thua */

/* flex-shrink: ty le thu nho khi thieu khong gian */
.khong-thu { flex-shrink: 0; } /* khong bao gio thu nho */
.thu-nhieu { flex-shrink: 2; } /* thu nho gap doi */

/* flex-basis: kich thuoc ban dau */
.co-dinh { flex-basis: 200px; }
.phan-tram { flex-basis: 33.333%; }

/* Shorthand flex: grow shrink basis */
.col { flex: 1 1 0%; }      /* grow=1, shrink=1, basis=0 */
.co-dinh { flex: 0 0 200px; } /* khong thay doi kich thuoc */
.tu-do { flex: 1; }          /* grow=1, shrink=1, basis=0 */
