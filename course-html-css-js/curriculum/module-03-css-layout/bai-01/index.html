Bai 01: Display block inline inline-block

Module 3 CSS Layout & Responsive

/* Block: chiem het chieu rong, xuong dong */
div, p, h1, section { display: block; }

/* Inline: chi rong bang noi dung, khong xuong dong */
span, a, strong { display: inline; }
/* Khong set width/height cho inline */

/* Inline-block: nhu inline nhung co the set width height */
.icon-btn { display: inline-block; width: 40px; height: 40px; }

/* None: an phan tu (khong chiem chu vi khong gian) */
.hidden { display: none; }
