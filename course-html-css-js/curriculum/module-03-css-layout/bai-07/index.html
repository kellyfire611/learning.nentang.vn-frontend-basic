Bai 07: Overflow hidden scroll auto

Module 3 CSS Layout & Responsive

/* Clip noi dung tran */
.clip { overflow: hidden; }

/* Thanh cuon (luon hien) */
.scroll { overflow: scroll; }

/* Thanh cuon (chi hien khi can) */
.hop-long {
  height: 200px;
  overflow: auto;  /* khuyen dung */
}

/* Chi cuon theo 1 huong */
.ngang { overflow-x: auto; overflow-y: hidden; }
.doc   { overflow-y: auto; overflow-x: hidden; }

/* Ung dung: table responsive */
.table-wrap { overflow-x: auto; }  /* table cuon ngang */
.table-wrap table { min-width: 600px; }

/* Clip text tran */
.ellipsis {
  white-space: nowrap;
  overflow: hidden;
  text-overflow: ellipsis;  /* hien ... */
}
