Bai 17: CSS Grid rows va gap

.grid {
  display: grid;
  grid-template-columns: 1fr 1fr 1fr;
  grid-template-rows: 100px auto 60px;
  gap: 20px; /* tat ca huong */
  /* hoac: row-gap: 16px; column-gap: 24px; */
}

/* Chieu cao hang tu dong */
.tu-dong { grid-auto-rows: minmax(100px, auto); }

/* Dat hang cu the */
.banner { grid-row: 1; grid-column: 1 / -1; } /* toan bo hang dau */

/* Implicit grid - hang tu dong tao them */
.grid {
  grid-auto-rows: 150px; /* hang tu dong cao 150px */
  grid-auto-flow: row;   /* mac dinh, them hang theo hang */
}
