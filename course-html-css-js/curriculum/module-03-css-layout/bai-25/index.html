Bai 25: Responsive Grid Layout

/* MAC DINH: 1 cot tren mobile */
.product-grid {
  display: grid;
  gap: 20px;
  /* khong can grid-template-columns - mac dinh 1 cot */
}

/* TABLET: 2 cot */
@media (min-width: 540px) {
  .product-grid {
    grid-template-columns: repeat(2, 1fr);
  }
}

/* DESKTOP NHO: 3 cot */
@media (min-width: 900px) {
  .product-grid {
    grid-template-columns: repeat(3, 1fr);
  }
}

/* DESKTOP LON: 4 cot */
@media (min-width: 1200px) {
  .product-grid {
    grid-template-columns: repeat(4, 1fr);
  }
}

/* HOAC: Dung auto-fill (an hoa: khong can media queries!) */
.smart-grid {
  display: grid;
  grid-template-columns: repeat(auto-fill, minmax(250px, 1fr));
  gap: 20px;
}
