Bai 23: Mobile-first Design

/* MOBILE FIRST: Bat dau tu mobile nho nhat, them dan len */

/* Base (mobile): tat ca style cho < 640px */
.container { padding: 0 16px; }
.card-grid { display: grid; gap: 16px; }
/* Mac dinh: 1 cot */

/* Tablet: tu 640px */
@media (min-width: 640px) {
  .container { max-width: 720px; margin: auto; }
  .card-grid { grid-template-columns: repeat(2, 1fr); } /* 2 cot */
}

/* Desktop: tu 1024px */
@media (min-width: 1024px) {
  .container { max-width: 1200px; }
  .card-grid { grid-template-columns: repeat(4, 1fr); } /* 4 cot */
}

/* Large Desktop: tu 1440px */
@media (min-width: 1440px) {
  .container { max-width: 1400px; }
}
