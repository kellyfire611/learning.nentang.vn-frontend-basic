Bai 18: CSS Grid column va row span

.grid {
  display: grid;
  grid-template-columns: repeat(4, 1fr);
  gap: 16px;
}

/* Du lieu nam o cot 1 den cot 3 */
.banner { grid-column: 1 / 3; }      /* cot 1 va 2 */
.banner { grid-column: 1 / span 2; } /* cung ket qua */
.banner { grid-column: 1 / -1; }     /* tu cot 1 den cuoi */

/* Du lieu nam o hang 1 den hang 3 */
.sidebar { grid-row: 1 / 3; }        /* keo dai 2 hang */

/* Phong to ca 2 chieu */
.featured {
  grid-column: span 2;
  grid-row: span 2;
}
