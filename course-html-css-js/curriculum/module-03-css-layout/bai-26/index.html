Trang Chu Module 3 Bai 26

Bai 26: Responsive Typography

/* Phuong phap 1: clamp() - co dinh min va max */
h1 { font-size: clamp(1.5rem, 5vw, 3rem); }
p  { font-size: clamp(0.9rem, 2vw, 1.1rem); }

/* Phuong phap 2: Media queries */
h1 { font-size: 1.8rem; }
@media (min-width: 640px)  { h1 { font-size: 2.2rem; } }
@media (min-width: 1024px) { h1 { font-size: 3rem; } }

/* Phuong phap 3: Vw (viewport width) */
h1 { font-size: 5vw; } /* 5% chieu rong man hinh */
/* Nhuoc diem: qua nho tren mobile, qua lon tren widescreen */

/* Fluid Typography - ket hop vw va clamp */
:root {
  --font-base: clamp(0.9rem, 0.85rem + 0.25vw, 1rem);
  --font-lg:   clamp(1.1rem, 1rem + 0.5vw, 1.25rem);
  --font-xl:   clamp(1.5rem, 1rem + 2.5vw, 2.5rem);
}
