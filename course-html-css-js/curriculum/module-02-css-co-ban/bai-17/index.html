Yeu cau va Code mau:

/* ::before va ::after - them noi dung trang tri */
/* Luon can "content" de hien thi */

/* Them bieu tuong truoc tieu de */
h2::before {
  content: " ";
}

/* Them gach ngang dep */
.tieu-de::after {
  content: "";
  display: block;
  width: 60px;
  height: 3px;
  background: #5a67d8;
  margin-top: 8px;
}

/* Tao chuyen dan */
a::after { content: " "; }

/* Tao badge "HOT" */
.san-pham::before {
  content: "HOT";
  position: absolute;
  top: 8px; right: 8px;
  background: red;
  color: white;
  font-size: 10px;
  padding: 2px 6px;
  border-radius: 10px;
}

/* Tao overlay trang tri */
.hinh-dep::after {
  content: "";
  position: absolute;
  inset: 0;
  background: rgba(90,103,216,0.2);
  transition: opacity 0.3s;
  opacity: 0;
}
.hinh-dep:hover::after { opacity: 1; }
