Bai 26: CSS Animation keyframes

/* Khai bao animation */
@keyframes nhay-len {
  0%   { transform: translateY(0); }
  50%  { transform: translateY(-20px); }
  100% { transform: translateY(0); }
}

@keyframes xoay {
  from { transform: rotate(0deg); }
  to   { transform: rotate(360deg); }
}

@keyframes mo-dan {
  from { opacity: 0; transform: translateY(20px); }
  to   { opacity: 1; transform: translateY(0); }
}

@keyframes nhip-tim {
  0%, 100% { transform: scale(1); }
  50%       { transform: scale(1.2); }
}

/* Ap dung animation */
.hieu-ung-nhay  { animation: nhay-len 0.6s ease infinite; }
.hieu-ung-xoay  { animation: xoay 2s linear infinite; }
.hieu-ung-mo    { animation: mo-dan 0.5s ease forwards; }
.hieu-ung-tim   { animation: nhip-tim 1s ease infinite; }
