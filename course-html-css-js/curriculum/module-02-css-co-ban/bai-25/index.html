Bai 25: CSS Transition

.hop { width:100px; height:100px; background:#5a67d8; border-radius:8px; }

/* Transition co ban */
.hop { transition: all 0.3s ease; }
.hop:hover { width:150px; background:#e53e3e; border-radius:50%; }

/* Transition tung thuoc tinh */
.btn {
  transition:
    background-color 0.2s ease,
    transform 0.15s ease,
    box-shadow 0.2s ease;
}

/* Cac ham timing */
.ease      { transition-timing-function: ease; }       /* mac dinh */
.linear    { transition-timing-function: linear; }     /* deu deu */
.ease-in   { transition-timing-function: ease-in; }    /* bat dau cham */
.ease-out  { transition-timing-function: ease-out; }   /* ket thuc cham */
.bounce    { transition-timing-function: cubic-bezier(0.68,-0.55,0.265,1.55); }

/* Delay */
.tre { transition: all 0.3s ease 0.5s; } /* tre 0.5 giay */
