/* Vi du tinh specificity */
p { color: black; }                /* 0,0,0,1 = 1 diem */
.text { color: blue; }             /* 0,0,1,0 = 10 diem */
#title { color: green; }           /* 0,1,0,0 = 100 diem */
#title.text p { color: red; }      /* 0,1,1,1 = 111 diem */

/* Tranh dung !important */
.van-ban { color: purple !important; } /* ghi de tat ca */
