Bài 28: Error Handling – try-catch

Module 4 – JavaScript Cơ Bản

📝 Yêu cầu bài tập

Xử lý lỗi trong JavaScript bằng try-catch-finally.

Mô tả

Lỗi không được xử lý khiến chương trình dừng đột ngột. try-catch bẫy lỗi để xử lý gracefully. finally luôn chạy dù có lỗi hay không.

Yêu cầu

  • Bẫy lỗi khi parse JSON không hợp lệ
  • Custom error class kế thừa Error
  • Xử lý lỗi trong async/await

🧠 Kiến thức cần nhớ

  • try { } catch(err) { } finally { }
  • err.message: thông báo lỗi
  • err.name: loại lỗi (TypeError, RangeError...)
  • throw new Error('message'): ném lỗi tự định nghĩa
  • finally: luôn chạy – dùng để cleanup (đóng file, loading spinner tắt)
  • Custom Error: class MyError extends Error { }

💻 Code mẫu

// try-catch cơ bản
try {
  const data = JSON.parse('{ sai json }');
} catch (err) {
  console.log('Lỗi JSON:', err.message);
  console.log('Loại lỗi:', err.name); // SyntaxError
}

// finally: luôn chạy – dùng để tắt loading
function fetchData(url) {
  showLoading(true);
  try {
    const data = getData(url);
    renderData(data);
  } catch (err) {
    showError(err.message);
  } finally {
    showLoading(false); // Luôn tắt loading
  }
}

// Custom Error
class ValidationError extends Error {
  constructor(field, message) {
    super(message);
    this.name = 'ValidationError';
    this.field = field;
  }
}
function validateAge(age) {
  if (age < 0 || age > 150) throw new ValidationError('age', 'Tuổi không hợp lệ');
  return true;
}
try { validateAge(-5); }
catch (err) {
  if (err instanceof ValidationError) console.log(`Lỗi trường ${err.field}: ${err.message}`);
  else throw err; // Re-throw lỗi không handle được
}
